class Solution {
public:
    void f(vector<vector<int>> &ans, vector<int> &temp, int k, int n, int i=1){
        if(i==10 || n==0){
            if(n==0 && temp.size()==k){
                ans.push_back(temp);
            }
            return;
        }
        if(i<=n){
            temp.push_back(i);
            f(ans,temp,k,n-i,i+1);
            temp.pop_back();
        }
        f(ans,temp,k,n,i+1);
    }
    vector<vector<int>> combinationSum3(int k, int n) {
        if(n==1) return {};
        vector<vector<int>> ans;
        vector<int> temp;
        f(ans,temp,k,n);
        return ans;
    }
};
