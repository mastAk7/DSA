class Solution {
public:
    int compress(vector<char>& f) {
        int j = 0;
        int n = f.size();
        for(int i=0; i<n; i++){
            int c = 1;
            while(i<n-1 && f[i]==f[i+1]){
                c++;
                i++;
            }
            f[j] = f[i];
            j++;
            if(c==1) continue;
            string temp;
            while(c){
                temp += ('0' + c%10);
                c/=10;
            }
            reverse(temp.begin(),temp.end());
            for(auto x:temp){
                f[j] = x;
                j++;
            }
        }
        while(j<f.size()) f.pop_back();
        return f.size();
    }
};
