class Solution {
public:
    string multiply(string s1, string s2) {
        int n1 = s1.size();
        int n2 = s2.size();
        string ans = "";
        string ze = "";
        for(int i=n2-1; i>=0; i--){
            int c = 0;
            int n = s2[i] - '0';
            if(i!=n2-1) ze+='0';
            string temp = ze;
            for(int j=n1-1; j>=0; j--){
                int x = (s1[j]-'0')*n + c;
                temp.push_back((char)('0'+x%10));
                c = x/10;
            }
            if(c) temp.push_back((char)('0'+c));
            int diff = temp.size() - ans.size();
            c = 0;
            for(int k=0; k<diff; k++) ans += '0';
            for(int k=0; k<temp.size(); k++){
                int x = (ans[k]-'0') + (temp[k]-'0') + c;
                ans[k] = (char)('0'+x%10);
                c = x/10;
            }
            if(c) ans += (char)('0'+c);
        }
        int n = ans.size()-1;
        while(n>0 && ans[n]=='0'){
            ans.pop_back();
            n--;
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
