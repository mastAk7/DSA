class Solution {
public:
    long long getDescentPeriods(vector<int>& prices) {
        int n = prices.size();
        long long ans = 0;
        for(int i=0; i<n; i++){
            long long c  = 1;
            int j = i+1;
            while(j<n && prices[j]==(prices[j-1]-1)){
                c++;
                j++;
            }
            i = j - 1;
            ans += (c*(c+1))/2;
        }
        return ans;
    }
};
