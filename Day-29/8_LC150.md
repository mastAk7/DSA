class Solution {
public:
    int f(string s){
        int c = 0, sign = s[0]=='-'? 1 : 0;
        for(int i=sign; i<s.size(); i++){ 
            c = c*10 + (s[i]-'0');
        }
        if(sign) return -c;
        return c;
    }
    int evalRPN(vector<string>& tokens) {
        vector<int> ans;
        for(auto x:tokens){
            char c = x[0];
            if(x.size()==1 && (c=='*' || c=='+' || c=='-' || c=='/')){
                int temp;
                int n = ans.size();
                if(c=='*') temp = ans[n-2]*ans[n-1];
                else if(c=='+') temp = ans[n-2]+ans[n-1];
                else if(c=='-') temp = ans[n-2]-ans[n-1];
                else temp = ans[n-2]/ans[n-1];
                ans.pop_back();
                ans.pop_back();
                ans.push_back(temp);
            } else ans.push_back(f(x));
        }
        return ans[0];
    }
};
