class Solution {
public:
    vector<int> findErrorNums(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(),nums.end());
        vector<int> ans;
        int sum = nums[nums.size()-1];
        for(int i=0; i<nums.size()-1; i++){
            if(nums[i]==nums[i+1]){
                ans.push_back(nums[i]);
            }
            sum += nums[i];
        }
        sum -= ans[0];
        ans.push_back((n*(n+1))/2 - sum);
        return ans;
    }
};
