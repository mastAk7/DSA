class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int ans = 0; 
        int i = 0;
        while(i<nums.size()){
            if(nums[i]==0){
                i++;
                continue;
            }
            int c = 1;
            while(i<nums.size()-1 && nums[i]==nums[i+1]){
                c++; 
                i++;
            }
            ans = max(ans,c);
            i++;
        }
        return ans;
    }
};
