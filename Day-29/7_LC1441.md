class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
        int i=1, j=0;
        int m = target.size();
        vector<string> ans;
        while(i<=n){
            ans.push_back("Push");
            if(target[j]==i) j++;
            else ans.push_back("Pop");
            if(j==m) break;
            i++;
        }
        return ans;
    }
};
