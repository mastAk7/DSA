class Solution {
public:
    vector<int> smallerNumbersThanCurrent(vector<int>& nums) {
        int mx = *max_element(nums.begin(),nums.end());
        vector<int> f(mx+2,0);
        for(int x:nums) f[x+1]++;
        for(int i=1; i<mx+2; i++) f[i] += f[i-1];
        for(int i=0; i<nums.size(); i++) nums[i] = f[nums[i]];
        return nums;
    }
};
