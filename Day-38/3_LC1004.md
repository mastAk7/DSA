class Solution {
public:
    int longestOnes(vector<int>& nums, int k) {
        int c = 0, ans = 0, l = 0, r = 0;
        while(r<nums.size()){
            if(!nums[r]){
                if(c==k){
                    while(l<=r && nums[l]) l++;
                    l++;
                } else c++;
            }
            ans = max(ans,r-l+1);
            r++;
        }
        return ans;
    }
};
