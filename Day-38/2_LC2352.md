class Solution {
public:
    string f(int n){
        string temp="";
        while(n){
            temp += '0' + n%10;
            n/=10; 
        }
        reverse(temp.begin(),temp.end());
        temp+='*';
        return temp;
    }
    int equalPairs(vector<vector<int>>& grid) {
        int n = grid.size();
        int ans = 0;
        unordered_map<string,int> mp;
        for(int i=0; i<n; i++){
            string temp;
            for(int j=0; j<n; j++){
                temp += (f(grid[i][j]));
            }
            cout<<temp<<endl;
            mp[temp]++;
        } 
        for(int i=0; i<n; i++){
            string temp;
            for(int j=0; j<n; j++){
                temp += f(grid[j][i]);
            }
            ans += mp[temp];
        } 
        return ans;
    }
};
