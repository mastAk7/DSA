class Solution {
public:
    long long maxPoints(vector<int>& f1, vector<int>& f2, int k) {
        map<int,vector<int>> mp;
        int n = f1.size();
        long long c = 0;
        for(int i=0; i<n; i++){
            long long x = f2[i] - f1[i];
            mp[x].push_back(i);
        }
        for(auto [x,y]:mp){
            for(int z:y){
                if(k){
                    c += (long long)f1[z];
                    k--;
                } else{
                    if(x<0) c+=(long long)f1[z];
                    else c+=(long long)f2[z];
                }
            }
        }
        return c;
    }
};
