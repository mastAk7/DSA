class Solution {
public:
    // the state is dp[i] -> max money robbed till i
    // there can be two cases:
    // 1. robber robs from the current house
    // 2. robber leaves this house
    // in the first case, we would consider max robbed till i-2 and add current money
    // this would ensure non adjacent houses robbed
    // in the second case, we would just take dp[i-1]
    // the max of both cases would be stored in dp[i]
    int rob(vector<int>& nums) {
        int n = nums.size();
        if(n==1) return nums[0];
        vector<int> dp(n);
        dp[0] = nums[0];
        dp[1] = max(dp[0],nums[1]);
        for(int i=2; i<n; i++){
            dp[i] = max(dp[i-1],dp[i-2]+nums[i]);
        }
        return dp[n-1];
    }
};
