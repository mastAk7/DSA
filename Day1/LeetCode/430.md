/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* prev;
    Node* next;
    Node* child;
};
*/

class Solution {
public:
    void f(Node* &head){
        if(!head->next){
            if(head->child){
                Node* temp = head->child;
                head->child = nullptr;
                head->next = temp;
                temp->prev = head;
                head = head->next;
                f(head);
            }
            return;
        }

        if(!head->child){
            head = head->next;
            f(head);
        } else{
            Node *tempNext = head->next;
            Node *tempChild = head->child;

            head->child = nullptr;
            head->next = tempChild;
            tempChild->prev = head;
            head = head->next;
            f(head);

            tempNext->prev = head;
            head->next = tempNext;
            head = head->next;
            f(head);
        }

        return;
    }
    Node* flatten(Node* head) {
        if(!head){
            return nullptr;
        }
        Node* temp = head;
        f(temp);
        return head;
    }
};
