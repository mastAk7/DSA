## Solution 1 -> Binary search

```cpp
typedef long long llm;

class Solution {
public:
    bool assistant(int m, int x){
        return (llm)m*m <= x;
    }
    int mySqrt(int x) {
        int  l = 0, r = x, ans = 0;
        while(l<=r){
            int m = (r-l)/2 + l;
            if(assistant(m,x)){
                ans = m;  // since we got the root, there might be a bigger number satisfying the condition -> we want the largest possible root
                l = m + 1;
            } else r = m - 1;
        }
        return ans;
    }
};
```

## Brute force

```cpp
class Solution {
public:
    int mySqrt(int x) {
        if(x==0 || x==1){
            return x;
        }
        for(int i=1; i<=x; i++){
            if(i>x/i){
                return i-1;
            }
        }
        return -1;
    }
};
```
