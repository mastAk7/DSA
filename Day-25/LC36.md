class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        for(int i=0; i<3; i++){
            for(int j=0; j<3; j++){
                vector<int> arr(9,0);
                for(int k=0; k<3; k++){
                    for(int l=0; l<3; l++){
                        if(board[i*3+k][j*3+l]=='.'){
                            continue;
                        }
                        int n = board[i*3+k][j*3+l] - '0' - 1;
                        if(arr[n]) return false;
                        arr[n]++;
                    }
                }
            }
        }
        
        for(int i=0; i<9; i++){
            vector<int> row(9,0), col(9,0);
            for(int j=0; j<9; j++){
                if(board[i][j]!='.'){
                    int n = board[i][j] - '0' - 1;
                    if(row[n]) return false;
                    row[n]++;
                }
                if(board[j][i]!='.'){
                    int n = board[j][i] - '0' - 1;
                    if(col[n]) return false;
                    col[n]++;
                }
            }
        }

        return true;
    }
};
