// state: dp[i] -> max length of lexo string till index i
// transition -> 
// for every row:
    // if an index < i is such that the element at i is greater
    // then dp[i] would be dp[index] + 1
    // this is because dp[index] stores the length of max lexo string
    // and since i is greater than index, all prior included elements 
    // must be greater
    // we would hence do this for every element < i and store the max

// base case: dp[i] = 1 because if we delete all other columns, the current one
// one would in lexo by default

// answer: since we do not know till column is deleted and which is taken,
// we take the max of dp

class Solution {
public:
    int minDeletionSize(vector<string>& s) {
        int n = s.size(), m = s[0].size();
        vector<int> dp(m,1);
        for(int i=1; i<m; i++){
            for(int j=0; j<i; j++){
                int check = 1;
                for(int row=0; row<n; row++){
                    if(s[row][i]<s[row][j]){
                        check = 0;
                        break;
                    }
                }
                if(check) dp[i] = max(dp[i],dp[j]+1);
            }
        }
        return m - *max_element(dp.begin(),dp.end());
    }
};
