class Solution {
public:
    vector<int> finalPrices(vector<int>& prices) {
        int n = prices.size();
        stack<int> s;
        
        for(int i = n-1; i >= 0; i--){
            while(!s.empty() && s.top() > prices[i])
                s.pop();
            
            int discount = !s.empty() ? s.top() : 0;
            s.push(prices[i]);
            prices[i] -= discount;
        }
        return prices;
    }
};
