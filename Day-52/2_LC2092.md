class Solution {
public:
    vector<int> findAllPeople(int n, vector<vector<int>>& meetings, int firstPerson) {
        sort(meetings.begin(), meetings.end(),[](auto &a, auto &b) {
            return a[2] < b[2];
        });

        set<int> know;
        know.insert(0);
        know.insert(firstPerson);

        int m = meetings.size();

        for (int i = 0; i < m; ) {
            int time = meetings[i][2];
            unordered_map<int, vector<int>> graph;

            int j = i;
            while (j < m && meetings[j][2] == time) {
                int u = meetings[j][0];
                int v = meetings[j][1];
                graph[u].push_back(v);
                graph[v].push_back(u);
                j++;
            }

            queue<int> q;
            unordered_set<int> visited;

            for (auto &p : graph) {
                if (know.count(p.first)) {
                    q.push(p.first);
                    visited.insert(p.first);
                }
            }

            while (!q.empty()) {
                int x = q.front(); q.pop();
                know.insert(x);

                for (int y : graph[x]) {
                    if (!visited.count(y)) {
                        visited.insert(y);
                        q.push(y);
                    }
                }
            }

            i = j;
        }

        return vector<int>(know.begin(), know.end());
    }
};
