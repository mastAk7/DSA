class Solution {
public:
    int minDeletionSize(vector<string>& s) {
        int n = s.size(), m = s[0].size();
        int i = 0;
        while(i<m){
            int j = 1;
            while(j<n){
                if(s[j][i]<s[j-1][i]){
                    i++;
                    break;
                }
                j++;
            }
            if(j==n) break;
        }
        if(i>=m-1) return i;
        int c = i;
        vector<int> f(m,0);
        for(int a = i+1; a<m; a++){
            for(int j=1; j<n; j++){
                if(s[j][i]!=s[j-1][i]) continue;
                for(int k=i+1; k<=a; k++){
                    if(f[k]) continue;
                    if(s[j][k]>s[j-1][k]) break;
                    if(s[j][k]<s[j-1][k]){
                        f[k] = 1;
                    }
                }
            }
        }
        for(int j = 0; j<m; j++) cout<<f[j]<<endl;
        for(int j = i+1; j<m; j++) if(f[j]) c++;
        return c;
    }
};
