class Solution {
public:
    bool cv(char c) {
        switch(tolower(c)) {
            case 'a': case 'e': case 'i': case 'o': case 'u':
                return true;
        }
        return false;
    }
    string sortVowels(string s) {
        vector<char> v;
        for(auto c:s) if(cv(c)) v.push_back(c);
        sort(v.rbegin(),v.rend());
        for(auto &c:s) if(cv(c)){
            c = v.back();
            v.pop_back();
        }
        return s;
    }
};
