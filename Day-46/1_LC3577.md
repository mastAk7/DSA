const int mod = 1e9 + 7;
class Solution {
public:
    int countPermutations(vector<int>& f) {
        int n = f.size();
        long long ans = 1;
        for(int i=1; i<n; i++){
            if(f[i]<=f[0]) return 0;
            ans *= i;
            ans %= mod;
        }
        return ans;
    }
};
