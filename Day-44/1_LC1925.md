class Solution {
public:
    int countTriples(int n) {
        int c = 0;
        for(int i=1; i<=n-2; i++){
            for(int j=i+1; j<=n-1; j++){
                double x = sqrt(j*j + i*i);
                if(floor(x) == x && x<=n) c+=2;
            }
        }
        return c;
    }
};
