## Method 1: sliding window
### TC:n , SC: 1
```cpp
class Solution {
public:
    int tribonacci(int n) {
        if(n==0) return 0;
        if(n<3) return 1;
        int a=0, b=1, c=1;
        for(int i=3; i<=n; i++){
            int temp = c;
            c += a+b;
            a = b;
            b = temp;
        }
        return c;
    }
};
```

## Method 2: DP
### TC:n , SC: n
```cpp
class Solution {
public:
    int tribonacci(int n) {
        if(n==0) return 0;
        if(n<3) return 1;
        vector<int> dp(n+1);
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 1;
        for(int i=3; i<=n; i++){
            dp[i] += dp[i-1] + dp[i-2] + dp[i-3];
        }
        return dp[n];
    }
};
```
